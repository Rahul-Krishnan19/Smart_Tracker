```python
def parse(self, email: dict) -> Optional[ParsedTransaction]:
    body = email.get("body", "")
    received_at = email.get("received_at")
    fallback_date = received_at.date() if received_at else date.today()
    body_clean = ' '.join(body.split())

    result = self._parse_upi_debit(body_clean, fallback_date)
    if result:
        return result

    result = self._parse_credit_card_debit(body_clean, fallback_date)
    if result:
        return result

    return None
```

def parse_email(email: dict) -> Optional[ParsedTransaction]:
    """Convenience: get parser and parse in one call. Returns None if no parser matches."""
    parser = get_parser(

        subject=email.get("subject", ""),
        body=email.get("body", ""),
