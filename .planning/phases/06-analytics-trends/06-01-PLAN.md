        1. Validate granularity; raise `ValueError` if not in `GRANULARITY_FORMAT`.
        2. Build a `base_query = db.query(Transaction).filter(Transaction.user_id == user_id)`. Apply date_from, date_to, payment_source filters identically to `transaction_service._build_filter_query` (date inclusive on both ends).
        3. `fmt = GRANULARITY_FORMAT[granularity]`. Group rows: `period_key = func.strftime(fmt, Transaction.transaction_date).label("period_key")`. Select `period_key, func.sum(amount), func.count(id), func.min(transaction_date), func.max(transaction_date)`. Group by `period_key`, order by `period_key`.
        4. For category_totals: a second query, `.with_entities(period_key, Transaction.category, func.sum(Transaction.amount))`, `.group_by(period_key, Transaction.category)`. Build a `dict[period_key, dict[category, float]]`.

           - `monthly`: `start.strftime("%b %Y")` → `"Apr 2026"`.

        9. `pct_change = self.compute_pct_change(current_total, previous_total)`.
        10. Return `{"trend": [...], "granularity": granularity, "current_total": current_total, "previous_total": previous_total, "pct_change": pct_change}`. Cast all `Decimal` to `float`.

        - Else: `return round((current - previous) / previous * 100, 2)`.

    ```python
    from datetime import date
    from typing import Optional
    from fastapi import APIRouter, Depends, Query, HTTPException
    from sqlalchemy.orm import Session

    from app.database import get_db
    from app.models.user import User
    from app.api.routes.auth import get_current_user
    from app.services.trend_service import trend_service, GRANULARITY_FORMAT
    from app.schemas.analytics import TrendResponse

    router = APIRouter()


    @router.get("/trend", response_model=TrendResponse)
    def get_trend(
        granularity: str = Query("monthly"),
        date_from: Optional[date] = Query(None),
        date_to: Optional[date] = Query(None),
        payment_source: Optional[str] = Query(None),
        current_user: User = Depends(get_current_user),
        db: Session = Depends(get_db),
    ):
        """ANA-03 / ANA-04 / ANA-05 / ANA-06: time-series trend with category totals
        and previous-period comparison."""
        if granularity not in GRANULARITY_FORMAT:
            raise HTTPException(
                status_code=422,
                detail=f"granularity must be one of {list(GRANULARITY_FORMAT.keys())}",
            )
        return trend_service.get_trend(
            db=db,
            user_id=current_user.id,
            granularity=granularity,
            date_from=date_from,
            date_to=date_to,
            payment_source=payment_source,
        )
    ```
